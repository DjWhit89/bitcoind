21struct ShortestVectorFirstComparator
22{
23 bool operator()(const std::vector<unsigned char>& a, const std::vector<unsigned char>& b) const
24 {
25 if (a.size() < b.size()) return true;
26 if (a.size() > b.size()) return false;
27 return a < b;
28 }
29};
