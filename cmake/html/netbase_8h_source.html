33enum class ConnectionDirection {
34 None = 0,
35 In = (1U << 0),
36 Out = (1U << 1),
37 Both = (In | Out),
38};
39static inline ConnectionDirection& operator|=(ConnectionDirection& a, ConnectionDirection b) {
40 using underlying = std::underlying_type_t<ConnectionDirection>;
41 a = ConnectionDirection(underlying(a) | underlying(b));
42 return a;
43}
44static inline bool operator&(ConnectionDirection a, ConnectionDirection b) {
45 using underlying = std::underlying_type_t<ConnectionDirection>;
46 return (underlying(a) & underlying(b));
47}
