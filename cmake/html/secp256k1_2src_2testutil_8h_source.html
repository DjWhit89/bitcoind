38static void testutil_random_fe_magnitude(secp256k1_fe *fe, int m) {
39 secp256k1_fe zero;
40 int n = testrand_int(m + 1);
41 secp256k1_fe_normalize(fe);
42 if (n == 0) {
43 return;
44 }
45 secp256k1_fe_set_int(&zero, 0);
46 secp256k1_fe_negate(&zero, &zero, 0);
47 secp256k1_fe_mul_int_unchecked(&zero, n - 1);
48 secp256k1_fe_add(fe, &zero);
49#ifdef VERIFY
50 CHECK(fe->magnitude == n);
51#endif
52}
