25// DoS prevention: limit cache size to 32MiB (over 1000000 entries on 64-bit
26// systems). Due to how we count cache size, actual memory usage is slightly
27// more (~32.25 MiB)
28static constexpr size_t DEFAULT_VALIDATION_CACHE_BYTES{32 << 20};
29static constexpr size_t DEFAULT_SIGNATURE_CACHE_BYTES{DEFAULT_VALIDATION_CACHE_BYTES / 2};
30static constexpr size_t DEFAULT_SCRIPT_EXECUTION_CACHE_BYTES{DEFAULT_VALIDATION_CACHE_BYTES / 2};
31static_assert(DEFAULT_VALIDATION_CACHE_BYTES == DEFAULT_SIGNATURE_CACHE_BYTES + DEFAULT_SCRIPT_EXECUTION_CACHE_BYTES);
